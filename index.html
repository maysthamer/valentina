// ═══════════════════════════════════════════════════════════════ // VALENTINA — Google Apps Script // يتوافق مع متغيرات الـ State في index.html (النسخة V3) // // طريقة التثبيت: // 1. افتح Google Sheets → Extensions → Apps Script // 2. احذف الكود الموجود والصق هذا الكود // 3. اضغط Deploy → New Deployment → Web App // 4. Execute as: Me | Who has access: Anyone // 5. انسخ الرابط وضعه في الإعدادات ← Google Sheets داخل الموقع // ═══════════════════════════════════════════════════════════════ // ── أسماء الشيتات (تتطابق مع State S في index.html) ────────── const SHEETS = { PRODUCTS: 'products', // S.products[] ORDERS: 'orders', // S.orders[] CUSTOMERS: 'customers', // S.customers[] BATCHES: 'batches', // S.batches[] SETTINGS: 'settings', // S.settings{} EXPENSES: 'expenses', // S.expenses{} }; // ── تهيئة الشيتات ──────────────────────────────────────────── function initSheets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const schemas = { // S.products[] fields [SHEETS.PRODUCTS]: [ 'id','name','category','ageGroup','sellPrice','buyPrice', 'barcode','desc','discType','discVal', 'sz_XS','sz_S','sz_M','sz_L','sz_XL','sz_XXL', 'sz_2Y','sz_3Y','sz_4Y','sz_5Y','sz_6Y','sz_8Y','sz_10Y','sz_12Y','sz_oneSize', 'createdAt','salesCount' ], // S.orders[] fields [SHEETS.ORDERS]: [ 'id','dateTs','date','status', 'customer_name','customer_phone','customer_addr','customer_notes', 'items_json', 'sub','ship','disc','total', 'offerMsg','offersLoss','promoUsed','rejectReason' ], // S.customers[] fields [SHEETS.CUSTOMERS]: [ 'id','name','phone','addr', 'orders','totalSpent','joinDate','lastOrder' ], // S.batches[] fields [SHEETS.BATCHES]: [ 'id','name','date','qty','totalCost','shipping','costPerPiece','itemsAdded','status' ], // S.expenses{} — كل صف = شهر [SHEETS.EXPENSES]: ['key','year','month','ads','ship','pack'], // S.settings{} [SHEETS.SETTINGS]: ['key','value'], }; Object.entries(schemas).forEach(([shName, headers]) => { let sheet = ss.getSheetByName(shName); if (!sheet) { sheet = ss.insertSheet(shName); const hr = sheet.getRange(1, 1, 1, headers.length); hr.setValues([headers]); hr.setBackground('#f2d5d5'); hr.setFontWeight('bold'); hr.setFontFamily('Tajawal'); sheet.setRightToLeft(true); sheet.setFrozenRows(1); } }); } // ══════════════════════════════════════════════════════════════ // GET — جلب كل البيانات دفعة واحدة // ══════════════════════════════════════════════════════════════ function doGet(e) { initSheets(); const action = e.parameter.action || 'getData'; const result = { products: getProducts(), orders: getOrders(), customers: getCustomers(), batches: getBatches(), expenses: getExpenses(), }; return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ══════════════════════════════════════════════════════════════ // POST — حفظ البيانات // ══════════════════════════════════════════════════════════════ function doPost(e) { initSheets(); let res = { success: false }; try { const data = JSON.parse(e.postData.contents); const action = data.action; if (action === 'addOrder') { addOrder(data.order); upsertCustomer(data.order.customer, data.order.total); res = { success: true, action: 'addOrder' }; } else if (action === 'updateProducts') { saveProducts(data.products); res = { success: true, action: 'updateProducts' }; } else if (action === 'addBatch') { addBatch(data.batch); res = { success: true, action: 'addBatch' }; } else if (action === 'saveExpenses') { saveExpenses(data.key, data.expenses); res = { success: true, action: 'saveExpenses' }; } else if (action === 'updateOrderStatus') { updateOrderStatus(data.orderId, data.status); res = { success: true, action: 'updateOrderStatus' }; } else if (action === 'test') { res = { success: true, action: 'test', message: 'Valentina Sheets Connected ✅' }; } } catch (err) { res = { success: false, error: err.toString() }; } return ContentService .createTextOutput(JSON.stringify(res)) .setMimeType(ContentService.MimeType.JSON); } // ══════════════════════════════════════════════════════════════ // PRODUCTS — S.products[] // الحقول: id, name, category, ageGroup, sellPrice, buyPrice, // barcode, desc, discType, discVal, // sizes: {XS,S,M,L,XL,XXL,2Y,3Y,...,one-size}, // createdAt, salesCount // ══════════════════════════════════════════════════════════════ function getProducts() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.PRODUCTS); if (!sheet || sheet.getLastRow() <= 1) return []; const data = sheet.getDataRange().getValues(); const headers = data[0]; return data.slice(1).filter(r => r[0]).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); // Rebuild sizes object from sz_* columns const sizes = {}; const szMap = { 'sz_XS':'XS','sz_S':'S','sz_M':'M','sz_L':'L','sz_XL':'XL','sz_XXL':'XXL', 'sz_2Y':'2Y','sz_3Y':'3Y','sz_4Y':'4Y','sz_5Y':'5Y', 'sz_6Y':'6Y','sz_8Y':'8Y','sz_10Y':'10Y','sz_12Y':'12Y', 'sz_oneSize':'one-size' }; Object.entries(szMap).forEach(([col, key]) => { if (obj[col] > 0) sizes[key] = parseInt(obj[col]) || 0; }); return { id: parseInt(obj.id), name: obj.name || '', category: obj.category || '', ageGroup: obj.ageGroup || null, sellPrice: parseFloat(obj.sellPrice) || 0, buyPrice: parseFloat(obj.buyPrice) || 0, barcode: obj.barcode || '', desc: obj.desc || '', discType: obj.discType || 'none', discVal: parseFloat(obj.discVal) || 0, sizes: sizes, createdAt: parseInt(obj.createdAt) || Date.now(), salesCount: parseInt(obj.salesCount) || 0, }; }); } function saveProducts(products) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEETS.PRODUCTS); const lastRow = sheet.getLastRow(); if (lastRow > 1) sheet.deleteRows(2, lastRow - 1); products.forEach(p => { const sizes = p.sizes || {}; sheet.appendRow([ p.id, p.name, p.category, p.ageGroup||'', p.sellPrice, p.buyPrice, p.barcode||'', p.desc||'', p.discType||'none', p.discVal||0, sizes['XS']||0, sizes['S']||0, sizes['M']||0, sizes['L']||0, sizes['XL']||0, sizes['XXL']||0, sizes['2Y']||0, sizes['3Y']||0, sizes['4Y']||0, sizes['5Y']||0, sizes['6Y']||0, sizes['8Y']||0, sizes['10Y']||0, sizes['12Y']||0, sizes['one-size']||0, p.createdAt||Date.now(), p.salesCount||0 ]); }); } // ══════════════════════════════════════════════════════════════ // ORDERS — S.orders[] // الحقول: id, dateTs, date, status, // customer: {name,phone,addr,notes}, // items: [{pid,sz,qty,price,name}], // sub, ship, disc, total, // offerMsg, offersLoss, promoUsed, rejectReason // ══════════════════════════════════════════════════════════════ function getOrders() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.ORDERS); if (!sheet || sheet.getLastRow() <= 1) return []; const data = sheet.getDataRange().getValues(); return data.slice(1).filter(r => r[0]).map(row => ({ id: parseInt(row[0]), dateTs: parseInt(row[1]) || parseInt(row[0]), date: row[2] || '', status: row[3] || 'preparing', customer: { name: row[4] || '', phone: String(row[5] || ''), addr: row[6] || '', notes: row[7] || '', }, items: JSON.parse(row[8] || '[]'), sub: parseFloat(row[9]) || 0, ship: parseFloat(row[10]) || 0, disc: parseFloat(row[11]) || 0, total: parseFloat(row[12]) || 0, offerMsg: row[13] || '-', offersLoss: parseFloat(row[14]) || 0, promoUsed: row[15] || null, rejectReason: row[16] || '', })); } function addOrder(order) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.ORDERS); sheet.appendRow([ order.id, order.dateTs || order.id, order.date || new Date().toLocaleDateString('ar-IQ'), order.status || 'preparing', order.customer?.name || '', order.customer?.phone || '', order.customer?.addr || '', order.customer?.notes || '', JSON.stringify(order.items || []), order.sub || 0, order.ship || 0, order.disc || 0, order.total || 0, order.offerMsg || '-', order.offersLoss || 0, order.promoUsed || '', order.rejectReason || '', ]); // تلوين الصف الجديد const lastRow = sheet.getLastRow(); sheet.getRange(lastRow, 1, 1, 17).setBackground('#fdf8f5'); } function updateOrderStatus(orderId, status) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.ORDERS); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (parseInt(data[i][0]) === parseInt(orderId)) { sheet.getRange(i + 1, 4).setValue(status); break; } } } // ══════════════════════════════════════════════════════════════ // CUSTOMERS — S.customers[] // الحقول: id, name, phone, addr, // orders, totalSpent, joinDate, lastOrder // ══════════════════════════════════════════════════════════════ function getCustomers() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.CUSTOMERS); if (!sheet || sheet.getLastRow() <= 1) return []; const data = sheet.getDataRange().getValues(); return data.slice(1).filter(r => r[0]).map(row => ({ id: parseInt(row[0]), name: row[1] || '', phone: String(row[2] || ''), addr: row[3] || '', orders: parseInt(row[4]) || 0, totalSpent: parseFloat(row[5]) || 0, joinDate: row[6] || '', lastOrder: row[7] || '', })); } function upsertCustomer(customer, total) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.CUSTOMERS); const data = sheet.getDataRange().getValues(); const today = new Date().toLocaleDateString('ar-IQ'); const phone = String(customer?.phone || ''); for (let i = 1; i < data.length; i++) { if (String(data[i][2]) === phone) { // تحديث عميل موجود sheet.getRange(i + 1, 5).setValue((parseInt(data[i][4]) || 0) + 1); sheet.getRange(i + 1, 6).setValue((parseFloat(data[i][5]) || 0) + (total || 0)); sheet.getRange(i + 1, 8).setValue(today); return; } } // إضافة عميل جديد sheet.appendRow([ Date.now(), customer?.name || '', phone, customer?.addr || '', 1, total || 0, today, today, ]); } // ══════════════════════════════════════════════════════════════ // BATCHES — S.batches[] // الحقول: id, name, date, qty, totalCost, shipping, // costPerPiece, itemsAdded, status // ══════════════════════════════════════════════════════════════ function getBatches() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.BATCHES); if (!sheet || sheet.getLastRow() <= 1) return []; const data = sheet.getDataRange().getValues(); return data.slice(1).filter(r => r[0]).map(row => ({ id: parseInt(row[0]), name: row[1] || '', date: row[2] || '', qty: parseInt(row[3]) || 0, totalCost: parseFloat(row[4]) || 0, shipping: parseFloat(row[5]) || 0, costPerPiece: parseFloat(row[6]) || 0, itemsAdded: parseInt(row[7]) || 0, status: row[8] || 'مكتمل', })); } function addBatch(batch) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.BATCHES); sheet.appendRow([ batch.id, batch.name, batch.date, batch.qty, batch.totalCost, batch.shipping, batch.costPerPiece, batch.itemsAdded, batch.status || 'مكتمل' ]); } // ══════════════════════════════════════════════════════════════ // EXPENSES — S.expenses{} // المفتاح: "year-month" // الحقول: {ads, ship, pack} // ══════════════════════════════════════════════════════════════ function getExpenses() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.EXPENSES); if (!sheet || sheet.getLastRow() <= 1) return {}; const data = sheet.getDataRange().getValues(); const result = {}; data.slice(1).forEach(row => { if (row[0]) result[row[0]] = { ads: row[3]||0, ship: row[4]||0, pack: row[5]||0 }; }); return result; } function saveExpenses(key, expenses) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.EXPENSES); const data = sheet.getDataRange().getValues(); const parts = key.split('-'); const year = parts[0], month = parts[1]; for (let i = 1; i < data.length; i++) { if (data[i][0] === key) { sheet.getRange(i + 1, 4, 1, 3).setValues([[expenses.ads||0, expenses.ship||0, expenses.pack||0]]); return; } } sheet.appendRow([key, year, month, expenses.ads||0, expenses.ship||0, expenses.pack||0]); }
